import pandas as pd
import numpy as np
from sklearn.cluster import KMeans
import statsmodels.api as sm
from statsmodels.formula.api import ols

# Sample Data (replace with your actual data loading)
data = {'FirmSize': [10, 20, 10, 30, 20, 30],
        'SectorLearningProportion': [0.2, 0.8, 0.2, 0.8, 0.2, 0.8],
        'PricingStrategy': [1.1, 1.9, 1.2, 2.8, 1.3, 2.9]}  # Example Pricing Strategy
df = pd.DataFrame(data)

# Clustering (example: K-Means)
kmeans = KMeans(n_clusters=3, random_state=0, n_init = 'auto') #Explicitly set n_init
df['Cluster'] = kmeans.fit_predict(df[['FirmSize', 'SectorLearningProportion', 'PricingStrategy']])

# Regression Analysis
model = ols("PricingStrategy ~ FirmSize + SectorLearningProportion + C(Cluster)", data=df).fit()
print(model.summary())

# Comparative Analysis (example: group means)
print(df.groupby('Cluster')[['FirmSize', 'SectorLearningProportion', 'PricingStrategy']].mean())
```
