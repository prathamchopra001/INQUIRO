## Example Code
```python
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the data
try:
    df = pd.read_csv("abm_data.csv") # Replace with your actual data file
except FileNotFoundError:
    print("Error: abm_data.csv not found. Ensure the file exists in the correct location.")
    exit()

# Calculate profit margin
df['profit_margin'] = (df['revenue'] - df['cost']) / df['revenue']

# Descriptive statistics
print(df['profit_margin'].describe())

# Group by sector and aggregate
sector_summary = df.groupby('sector')['profit_margin'].agg(['mean', 'median', 'std'])
print(sector_summary)

# Data visualization
plt.figure(figsize=(10, 6))
sns.boxplot(x='sector', y='profit_margin', data=df)
plt.title('Profit Margin Distribution by Sector')
plt.xlabel('Sector')
plt.ylabel('Profit Margin')
plt.xticks(rotation=45, ha='right') # Rotate x-axis labels for readability
plt.tight_layout() # Adjust layout to prevent labels from overlapping
plt.savefig("profit_margin_boxplot.png") # Save the figure
plt.close() # Close the figure to free memory
```
