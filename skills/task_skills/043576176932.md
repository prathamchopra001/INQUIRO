import numpy as np

class Firm(mesa.Agent):

        self.price = 10

        super().__init__(unique_id, model)

    def step(self):
        pass # Customers are passive in this example

class MarketModel(mesa.Model):
    def __init__(self, num_firms, num_customers):
        super().__init__()
        self.num_firms = num_firms
        self.num_customers = num_customers
        self.schedule = mesa.time.SimultaneousActivation(self)

        # Create firms with varying sizes

            customer = Customer(i, self)

            agent_reporters={"Price": "price", "Size": "size"}
        )

        self.datacollector.collect(self)

## Common Pitfalls
*   **Incorrect Probability Calculation:** Ensure probabilities for firm matching sum to 1.
*   **Lack of Data Collection:** Use `mesa.DataCollector` to track relevant variables (e.g., price, size).
*   **Inefficient Matching:** Optimize the matching logic for performance with large numbers of agents.
*   **Missing `step()` method**: Ensure `step()` method exists for all agents and the model.
*   **Forgetting to add agents to the scheduler**: All agents must be added using `self.schedule.add(agent)`.
