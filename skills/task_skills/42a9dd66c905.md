import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.arima.model import ARIMA

def simulate_abm(num_firms, learning_proportion, time_steps=100):
    """Simulates a simple ABM with learning and non-learning firms."""
    rng = np.random.default_rng()
    num_learning_firms = int(num_firms * learning_proportion)
    firm_forecasts = np.zeros((num_firms, time_steps))
    actual_values = np.random.randn(time_steps)  # Replace with real data generation

    for t in range(1, time_steps):
        # Firm forecasting logic (simplified)
        for i in range(num_firms):
            if i < num_learning_firms:
                # Learning firm (e.g., using ARIMA)
                try:
                    model = ARIMA(actual_values[:t], order=(1, 0, 0)) # Simple AR(1)
                    model_fit = model.fit()
                    firm_forecasts[i, t] = model_fit.forecast(steps=1)[0]
                except:
                    firm_forecasts[i, t] = np.mean(actual_values[:t]) # Fallback
            else:
                # Non-learning firm (e.g., naive forecast)
                firm_forecasts[i, t] = actual_values[t-1] if t > 0 else 0

    # Aggregate forecast (e.g., mean)
    aggregate_forecast = np.mean(firm_forecasts[:, 1:], axis=0)
    return actual_values[1:], aggregate_forecast

# Example usage
actual, forecast = simulate_abm(num_firms=100, learning_proportion=0.5)

# Basic plotting
plt.plot(actual, label="Actual")
plt.plot(forecast, label="Forecast")
plt.legend()
plt.show()

## Common Pitfalls
*   **Incorrect indexing:** ABMs often involve time series data; be careful about off-by-one errors.
*   **Data leakage:** Ensure learning firms only use past data for forecasting.
*   **Lack of Vectorization:** Prefer NumPy operations over Python loops for performance.
*   **Overfitting in Learning Models:** Use appropriate regularization or validation techniques when using more complex learning algorithms.
*   **Insufficient Time Steps:** Ensure the simulation runs for a sufficient duration to observe meaningful patterns.
*   **No Seed:** Set `numpy.random.default_rng(seed)` for reproducibility.
