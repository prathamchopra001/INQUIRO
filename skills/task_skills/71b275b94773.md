import mesa
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

class Firm(mesa.Agent):
    def __init__(self, unique_id, model, initial_price=10):
        super().__init__(unique_id, model)
        self.price = initial_price
        self.learning_rate = 0.01

    def step(self):
        # Simple price adjustment based on market conditions
        market_price = self.model.get_market_price()
        if market_price > self.price:
            self.price += self.learning_rate * (market_price - self.price)
        else:
            self.price -= self.learning_rate * (self.price - market_price)
        self.price = max(1, self.price) # Ensure price is positive

class MarketModel(mesa.Model):
    def __init__(self, num_firms, initial_price=10):
        super().__init__()
        self.num_firms = num_firms
        self.schedule = mesa.time.SimultaneousActivation(self)
        self.firms = []
        for i in range(self.num_firms):
            f = Firm(i, self, initial_price)
            self.schedule.add(f)
            self.firms.append(f)
        self.datacollector = mesa.DataCollector(
            model_reporters={"HHI": self.calculate_hhi, "AvgPrice": self.get_average_price},
            agent_reporters={"Price": "price"}
        )

    def step(self):
        self.schedule.step()
        self.datacollector.collect(self)

    def get_market_price(self):
        prices = [firm.price for firm in self.firms]
        return np.mean(prices)

    def calculate_hhi(self):
        prices = [firm.price for firm in self.firms]
        total_revenue = sum(prices)  #Simplified, assuming quantity=1
        market_shares = [(price / total_revenue)**2 for price in prices]
        return sum(market_shares) * 10000

    def get_average_price(self):
        prices = [firm.price for firm in self.firms]
        return np.mean(prices)
```

## Common Pitfalls
*   **Incorrect HHI Calculation:** Ensure you are using squared *market shares*, not just prices. Normalize prices to get market shares correctly.
*   **Data Collection Errors:** Use `mesa.DataCollector` correctly to store model and agent-level data.  Verify that the collected data is what you expect.
*   **Model Initialization:**  Ensure initial conditions (number of firms, initial prices) are correctly set.
*   **Price Convergence Issues:**  The model might converge to a single price quickly. Adjust learning rates or add noise to firm behavior for more interesting dynamics.
*   **Missing Visualization:** Remember to visualize HHI and average price over time to analyze the impact of market concentration. Use `matplotlib` or similar.
