```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import gym
from gym import spaces

class PricingEnvironment(gym.Env):
    def __init__(self, n_actions=3, initial_price=10, price_range=5):
        super(PricingEnvironment, self).__init__()
        self.action_space = spaces.Discrete(n_actions)  # Example: 0=lower, 1=same, 2=higher
        self.observation_space = spaces.Box(low=0, high=100, shape=(1,), dtype=np.float32) # Simplified state (current price)
        self.current_price = initial_price
        self.price_range = price_range
        self.time_step = 0
        self.max_steps = 100

    def step(self, action):
        self.time_step += 1
        if action == 0:
            self.current_price = max(1, self.current_price - 1) #Prevent negative prices
        elif action == 2:
            self.current_price = self.current_price + 1
        # action == 1 does nothing

        # Simplified reward function (example)
        reward = self.current_price # Higher price is better (initially)
        if self.time_step > 50 and self.current_price > self.price_range + 5: #Penalty for high price
          reward = -1 * self.current_price

        done = self.time_step >= self.max_steps
        truncated = False #Deprecated in newer gym versions.
        info = {}
        return np.array([self.current_price], dtype=np.float32), reward, done, truncated, info

    def reset(self, seed=None, options=None):
        super().reset(seed=seed, options=options)
        self.current_price = 10
        self.time_step = 0
        return np.array([self.current_price], dtype=np.float32), {}

def simple_q_learning(env, learning_rate=0.1, discount_factor=0.9, episodes=100):
    q_table = np.zeros((env.observation_space.shape[0] * 100, env.action_space.n)) #Very simplified Q table (price * 100 as index)
    rewards_per_episode = []

    for episode in range(episodes):
        state, _ = env.reset()
        total_reward = 0
        done = False
        truncated = False
        while not done and not truncated:
            action = np.argmax(q_table[int(state[0]*100), :]) #Exploitation only
            new_state, reward, done, truncated, _ = env.step(action)

            q_table[int(state[0]*100), action] = q_table[int(state[0]*100), action] + learning_rate * (reward + discount_factor * np.max(q_table[int(new_state[0]*100), :]) - q_table[int(state[0]*100), action])
            state = new_state
            total_reward += reward
        rewards_per_episode.append(total_reward)
    return q_table, rewards_per_episode

# Example Usage:
env = PricingEnvironment()
q_table, rewards = simple_q_learning(env, learning_rate=0.2)

plt.plot(rewards)
plt.xlabel("Episode")
plt.ylabel("Total Reward")
plt.title("Q-learning Rewards")
plt.show()

print("Final Q-Table:\n", q_table)
```

## Common Pitfalls
*   **Incorrect State Representation:**  The environment's `observation_space` and the Q-table's state representation must align. Mismatch leads to indexing errors and poor learning.
*   **Exploration vs. Exploitation:**  The example is purely exploitation. Implement an epsilon-greedy strategy for better exploration.
*   **Learning Rate Sensitivity:** The optimal learning rate varies greatly. Experiment with values between 0.001 and 0.5.
*   **Environment Complexity:** The `PricingEnvironment` is highly simplified. A real-world environment requires more sophisticated state and reward functions, including demand curves, competitor pricing, and time-varying factors.
*   **Non-Stationary Environment**: The environment may change over time (e.g. changing demand). This requires adaptive learning rates or other techniques to handle non-stationarity.
