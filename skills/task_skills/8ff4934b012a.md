## Example Code
```python
import mesa
import numpy as np

class Firm(mesa.Agent):
    def __init__(self, unique_id, model, size):
        super().__init__(unique_id, model)
        self.size = size
        self.matched = False

    def step(self):
        # Probability of matching increases with size
        match_probability = min(self.size / self.model.total_size, 1.0)  # Ensure probability <= 1

        if self.random.random() < match_probability:
            self.matched = True
        else:
            self.matched = False


class Customer(mesa.Agent):
    def __init__(self, unique_id, model):
        super().__init__(unique_id, model)
        self.matched = False

    def step(self):
        pass # Customers passively wait to be matched

class MatchingModel(mesa.Model):
    def __init__(self, num_firms, num_customers, firm_sizes):
        self.num_firms = num_firms
        self.num_customers = num_customers
        self.schedule = mesa.time.SimultaneousActivation(self)
        self.total_size = sum(firm_sizes)

        # Create firms
        for i in range(self.num_firms):
            firm = Firm(i, self, firm_sizes[i])
            self.schedule.add(firm)

        # Create customers
        for i in range(self.num_customers):
            customer = Customer(i + num_firms, self)
            self.schedule.add(customer)


    def step(self):
        self.schedule.step()


# Example usage
model = MatchingModel(num_firms=5, num_customers=10, firm_sizes=[1, 2, 3, 4, 5])
for _ in range(5):
    model.step()

# Access agent data (example: check which firms matched)
for agent in model.schedule.agents:
    if isinstance(agent, Firm):
        print(f"Firm {agent.unique_id} matched: {agent.matched}")
```

*   **Not Updating Model State:**  Ensure the `step()` function updates the agents' states (e.g., `matched`) appropriately.
