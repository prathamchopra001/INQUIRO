import numpy as np
import pandas as pd
from sklearn.metrics import mean_squared_error
import statsmodels.api as sm
import random

def simulate_abm(num_firms, learning_proportion, time_steps=100):
    """Simulates a simplified agent-based model."""

    # Initialize GDP, inflation, investment, household consumption, government consumption (simplified)
    gdp_history = [100]
    inflation_history = [2]
    investment_history = [20]
    consumption_history = [60]
    government_history = [20]

    # Determine number of learning firms
    num_learning_firms = int(num_firms * learning_proportion)

    # Simulation Loop
    for t in range(1, time_steps):
        # Simplified forecasting (example)
        forecasted_gdp = gdp_history[-1] * (1 + 0.02 * (1 + random.uniform(-0.01,0.01))) # Random growth

        # Update economic indicators (very simplified)
        gdp_history.append(forecasted_gdp)
        inflation_history.append(inflation_history[-1] * (1 + random.uniform(-0.005, 0.005)))
        investment_history.append(investment_history[-1] * (1 + random.uniform(-0.01, 0.01)))
        consumption_history.append(consumption_history[-1] * (1 + random.uniform(-0.01, 0.01)))
        government_history.append(government_history[-1] * (1 + random.uniform(-0.001, 0.001)))

    # Create a Pandas DataFrame for easy analysis
    df = pd.DataFrame({
        'GDP': gdp_history,
        'Inflation': inflation_history,
        'Investment': investment_history,
        'Consumption': consumption_history,
        'Government': government_history
    })
    return df


# Example Usage
num_firms = 100
learning_proportion = 0.5
results_df = simulate_abm(num_firms, learning_proportion)

# Example forecasting performance measurement (using simple persistence model)
actual_gdp = results_df['GDP'][1:].values
forecasted_gdp = results_df['GDP'][:-1].values  # Lagged GDP as forecast

rmse = np.sqrt(mean_squared_error(actual_gdp, forecasted_gdp))
print(f"RMSE for GDP: {rmse}")
