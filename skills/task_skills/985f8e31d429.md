class Firm:
    def __init__(self, firm_id, initial_price, learning_rate=0.1, discount_factor=0.9, epsilon=0.1):
        self.firm_id = firm_id
        self.price = initial_price
        self.learning_rate = learning_rate
        self.discount_factor = discount_factor
        self.epsilon = epsilon
        self.q_table = {} # {(state, action): value}

    def get_state(self, relative_profitability):
        # Discretize relative profitability into bins
        if relative_profitability < -0.1: return "low"
        elif relative_profitability > 0.1: return "high"
        else: return "medium"

    def get_possible_actions(self):
        return [-0.05, 0, 0.05]  # Decrease, no change, Increase price by 5%

    def choose_action(self, state):
        if random.uniform(0, 1) < self.epsilon:
            return random.choice(self.get_possible_actions())
        else:
            actions = self.get_possible_actions()
            q_values = [self.q_table.get((state, action), 0) for action in actions] # Default Q-value is 0
            max_q = max(q_values)
            best_actions = [actions[i] for i, q in enumerate(q_values) if q == max_q]
            return random.choice(best_actions)

    def update_q_table(self, state, action, reward, next_state):
        old_value = self.q_table.get((state, action), 0)
        next_max = max([self.q_table.get((next_state, a), 0) for a in self.get_possible_actions()]) if next_state else 0
        new_value = (1 - self.learning_rate) * old_value + self.learning_rate * (reward + self.discount_factor * next_max)
        self.q_table[(state, action)] = new_value

    def adjust_price(self, action):
        self.price *= (1 + action)
        self.price = max(0.1, self.price) # Ensure price doesn't go below 0.1

## Common Pitfalls
*   **Incorrect Q-table initialization:** Ensure Q-values are initialized (e.g., to 0) when first accessed.  Using `.get(key, 0)` is crucial.
*   **Exploitation vs. Exploration:** Tune `epsilon` to balance exploration (trying new actions) and exploitation (using the best-known action).  Start with a higher value (e.g., 0.1) and potentially decay it over time.
*   **State Space Definition:** The `get_state` function is critical. Ensure the state space is granular enough to capture relevant information but not so large that learning becomes impossible.
*   **Reward Function:** The reward function should be carefully designed to incentivize desired behavior (e.g., higher relative profits).
*   **Price Boundaries:** Prevent prices from becoming negative or excessively large.
