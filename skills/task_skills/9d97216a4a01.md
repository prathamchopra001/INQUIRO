import mesa

class Firm(mesa.Agent):
    def __init__(self, unique_id, model, size):
        super().__init__(unique_id, model)
        self.size = size
        self.price = 10  # Initial price
        self.profit = 0

    def step(self):
        # Example: Adjust price randomly
        self.price += np.random.uniform(-1, 1)
        self.price = max(1, self.price) # Ensure price doesn't drop too low

    def match(self, other_agent):
        # Larger firms have higher matching probability
        prob = self.size / (self.size + other_agent.size) # Simple example
        if self.random.random() < prob:
            # Simulate a transaction
            revenue = self.price # Simplified for example
            cost = self.size # Simplified, cost scales with size
            self.profit += revenue - cost
            return True
        return False

class Model(mesa.Model):
    def __init__(self, num_firms, initial_size_range):
        super().__init__()
        self.num_firms = num_firms
        self.schedule = mesa.time.RandomActivation(self)

        for i in range(self.num_firms):
            size = np.random.randint(initial_size_range[0], initial_size_range[1])
            firm = Firm(i, self, size)
            self.schedule.add(firm)

    def step(self):
        # Matching logic (simplified example)
        firms = self.schedule.agents
        for i in range(len(firms)):
            for j in range(i + 1, len(firms)):
                firms[i].match(firms[j])
                firms[j].match(firms[i]) # Ensure both directions are considered

        self.schedule.step()


# Example Run
model = Model(num_firms=10, initial_size_range=(1, 5))
for i in range(10):
    model.step()

# Example analysis
profits = [agent.profit for agent in model.schedule.agents]
print(f"Average Profit: {np.mean(profits)}")

## Common Pitfalls
*   **Incorrect Matching Logic:** Ensure matching is symmetric (if A matches B, B also has a chance to match A) or asymmetric as designed.
*   **Ignoring Firm Size in Matching Probability:** Firm size should directly influence the `prob` calculation in the `match` method.
*   **Poor Data Collection:** Collect price and profit data at each step for each agent using the DataCollector.
*   **Inefficient Matching Algorithm:** For large numbers of agents, optimize the matching process (e.g., spatial indexing). The provided example uses a naive O(n^2) approach.
*   **Lack of Heterogeneity Analysis:** Analyze profit and pricing *across* different firm sizes, not just averages. Use Pandas `groupby` or similar techniques.
*   **Not Visualizing Results:** Use `matplotlib` and `seaborn` to visualize profit distributions, price trends, and the relationship between firm size and performance.
