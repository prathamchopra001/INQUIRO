```python
import mesa
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy.stats import variation

class Firm(mesa.Agent):
    def __init__(self, unique_id, model):
        super().__init__(unique_id, model)
        self.price = 10  # Initial price

    def step(self):
        # Simple price adjustment (example)
        self.price += np.random.normal(0, 0.1)
        self.price = max(1, self.price) # Ensure price is positive

class MarketModel(mesa.Model):
    def __init__(self, num_firms, random_seed=None):
        super().__init__(random_seed=random_seed)
        self.num_firms = num_firms
        self.schedule = mesa.time.SimultaneousActivation(self)
        for i in range(num_firms):
            firm = Firm(i, self)
            self.schedule.add(firm)
        self.datacollector = mesa.DataCollector(
            model_reporters={"MeanPrice": lambda m: np.mean([a.price for a in m.schedule.agents])},
            agent_reporters={"Price": "price"}
        )

    def step(self):
        self.datacollector.collect(self)
        self.schedule.step()

def run_simulation(num_firms, num_steps, seed):
    model = MarketModel(num_firms=num_firms, random_seed=seed)
    for _ in range(num_steps):
        model.step()
    return model.datacollector.get_model_vars_dataframe()

# Example usage:
num_simulations = 5 # Reduced for brevity
results = []
for i in range(num_simulations):
    df = run_simulation(num_firms=10, num_steps=100, seed=i)
    df['simulation'] = i
    results.append(df)

all_results = pd.concat(results)

# Analyze variability (example)
final_prices = all_results.groupby('simulation')['MeanPrice'].last()
price_std = np.std(final_prices)
price_mean = np.mean(final_prices)
price_cv = variation(final_prices)

print(f"Mean Final Price: {price_mean}")
print(f"Std Dev of Final Price: {price_std}")
print(f"Coefficient of Variation of Final Price: {price_cv}")

# Visualization (example)
plt.figure()
all_results.groupby('simulation')['MeanPrice'].plot()
plt.xlabel("Step")
plt.ylabel("Mean Price")
plt.title("Mean Price over Simulations")
plt.show()
```

## Common Pitfalls
*   **Incorrect Seed Handling**: Ensure the `random_seed` is correctly passed to the model and used within the agent's behavior.  Different seeds *must* produce different results.
*   **Insufficient Data Collection**: Collect relevant data (agent-level and model-level) for analysis.
*   **Ignoring Burn-in Period**: Discard initial simulation steps as they might not be representative of the long-term dynamics.
*   **Misinterpreting Statistical Measures**: Use appropriate statistical measures (e.g., standard deviation, coefficient of variation) to quantify variability.
*   **Poor Visualization**: Create clear and informative visualizations to communicate the results effectively. Use labels and titles.
*   **Not Vectorizing**: Use numpy vectorization where possible instead of looping.
