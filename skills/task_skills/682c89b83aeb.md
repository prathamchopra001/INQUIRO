```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import gym
from gym import spaces
import random

class PricingEnv(gym.Env):
    def __init__(self):
        super(PricingEnv, self).__init__()
        self.observation_space = spaces.Discrete(10)  # Example: 10 price levels
        self.action_space = spaces.Discrete(3)       # Example: Increase, Decrease, Maintain
        self.inventory = 100
        self.price = 5

    def step(self, action):
        # Example simplistic demand function
        demand = max(0, 20 - abs(self.price - 5) * 5 + random.randint(-5, 5))
        demand = min(demand, self.inventory)  # Cannot sell more than available

        if action == 0:  # Increase price
            self.price += 1
        elif action == 1:  # Decrease price
            self.price -= 1
        # else: Maintain price

        reward = self.price * demand
        self.inventory -= demand
        done = self.inventory <= 0

        observation = int(self.price) % 10 # Simplified observation

        return observation, reward, done, {}

    def reset(self):
        self.inventory = 100
        self.price = 5
        return int(self.price) % 10

def rl_agent(env, learning_rate=0.1, discount_factor=0.9, exploration_rate=0.1, episodes=100):
    q_table = np.zeros((env.observation_space.n, env.action_space.n))
    rewards_per_episode = []

    for episode in range(episodes):
        state = env.reset()
        done = False
        total_reward = 0

        while not done:
            if random.uniform(0, 1) < exploration_rate:
                action = env.action_space.sample() # Explore
            else:
                action = np.argmax(q_table[state, :]) # Exploit

            new_state, reward, done, _ = env.step(action)

            q_table[state, action] = q_table[state, action] + learning_rate * (reward + discount_factor * np.max(q_table[new_state, :]) - q_table[state, action])

            total_reward += reward
            state = new_state
        rewards_per_episode.append(total_reward)

    return q_table, rewards_per_episode

# Sensitivity Analysis
learning_rates = [0.01, 0.1, 0.5]
all_rewards = {}

env = PricingEnv() # Instantiate environment outside the loop

for lr in learning_rates:
    q_table, rewards = rl_agent(env, learning_rate=lr, episodes=100)
    all_rewards[lr] = rewards

# Plotting the results
for lr, rewards in all_rewards.items():
    plt.plot(rewards, label=f"Learning Rate: {lr}")

plt.xlabel("Episode")
plt.ylabel("Total Reward")
plt.title("Sensitivity Analysis of Learning Rate")
plt.legend()
plt.show()

```

## Common Pitfalls
*   **Incorrect Q-Table Update:** Double-check the Bellman equation implementation.
*   **Ignoring Environment Dynamics:** The `PricingEnv` is simplistic; real-world demand is more complex.
*   **Insufficient Episodes:** Ensure enough episodes for convergence.
*   **Unstable Learning:** High learning rates can lead to instability. Monitor reward curves.
*   **No Exploration:** Without exploration (epsilon-greedy), the agent might get stuck in local optima.
*   **Incorrect Observation/Action Space Definition:** Ensure that these match the environment dynamics.
*   **Not resetting the environment correctly:** The `reset()` method is vital before each episode.
