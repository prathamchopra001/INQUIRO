import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import random

class SimpleEnvironment:
    def __init__(self, states=5, actions=2):
        self.states = states
        self.actions = actions
        self.current_state = 0

    def step(self, action, reward_function):
        if action == 0:
            self.current_state = max(0, self.current_state - 1)
        else:
            self.current_state = min(self.states - 1, self.current_state + 1)
        reward = reward_function(self.current_state)
        done = self.current_state == self.states - 1  # Example termination
        return self.current_state, reward, done

def q_learning(env, reward_function, episodes=100, learning_rate=0.1, discount_factor=0.9, epsilon=0.1):
    q_table = np.zeros((env.states, env.actions))
    rewards_per_episode = []

    for episode in range(episodes):
        state = 0 # Reset to initial state
        total_reward = 0
        done = False
        while not done:
            if random.uniform(0, 1) < epsilon:
                action = random.choice(range(env.actions))
            else:
                action = np.argmax(q_table[state, :])

            next_state, reward, done = env.step(action, reward_function)
            total_reward += reward

            best_next_q = np.max(q_table[next_state, :])
            q_table[state, action] += learning_rate * (reward + discount_factor * best_next_q - q_table[state, action])
            state = next_state
        rewards_per_episode.append(total_reward)
    return q_table, rewards_per_episode

# Example reward functions
def absolute_profit(state):
    return state  # Example: Profit increases with state

def relative_profit(state):
    return state / 4 # Example: Profit relative to max state

def market_share(state):
    return (state + 1) ** 2 # Example: Market share increases quadratically

# Run Q-learning with different reward functions
env = SimpleEnvironment()
q_table_abs, rewards_abs = q_learning(env, absolute_profit)
q_table_rel, rewards_rel = q_learning(env, relative_profit)
q_table_mkt, rewards_mkt = q_learning(env, market_share)

# Plotting (example - expand as needed)
plt.plot(rewards_abs, label="Absolute Profit")
plt.plot(rewards_rel, label="Relative Profit")
plt.plot(rewards_mkt, label="Market Share")
plt.xlabel("Episode")
plt.ylabel("Reward")
plt.legend()
plt.show()

# Basic statistical analysis (example)
print(f"Average reward (Absolute Profit): {np.mean(rewards_abs)}")
print(f"Average reward (Relative Profit): {np.mean(rewards_rel)}")
print(f"Average reward (Market Share): {np.mean(rewards_mkt)}")

## 5. Common Pitfalls
*   **Non-Stationary Environments:** Ensure the environment doesn't change drastically during training, which can hinder convergence.
*   **Incorrect Reward Scaling:** Reward functions with vastly different scales can lead to instability. Normalize or scale rewards appropriately.
*   **Insufficient Exploration:** The agent might get stuck in local optima if exploration (epsilon) is too low.
*   **Learning Rate Tuning:** The learning rate needs to be tuned based on the complexity of the environment and reward function.
*   **Discount Factor:** A high discount factor can propagate rewards too far, while a low factor makes the agent myopic.
*   **Environment Reset:** Always reset the environment to a starting state at the beginning of each episode.
*   **Convergence Metrics:** Use appropriate convergence metrics (e.g., moving average of rewards, Q-table stability) for quantitative analysis.
