```python
# Assuming 'df' is a pandas DataFrame with columns 'firm_size' and 'market_share' and 'year'
# Example: df = pd.read_csv('italian_firm_data.csv')

# Descriptive Statistics
print(df[['firm_size', 'market_share']].describe())

# Data Visualization
plt.figure(figsize=(10, 6))
sns.histplot(df['firm_size'], kde=True)
plt.title('Distribution of Firm Sizes')
plt.xlabel('Firm Size')
plt.ylabel('Frequency')
plt.show()

plt.figure(figsize=(10, 6))
sns.scatterplot(x='firm_size', y='market_share', data=df)
plt.title('Firm Size vs Market Share')
plt.xlabel('Firm Size')
plt.ylabel('Market Share')
plt.show()

# Time Series Analysis (if 'year' column exists)
if 'year' in df.columns:
    time_series_data = df.groupby('year')[['firm_size', 'market_share']].mean()
    plt.figure(figsize=(12, 6))
    plt.plot(time_series_data.index, time_series_data['firm_size'], label='Average Firm Size')
    plt.plot(time_series_data.index, time_series_data['market_share'], label='Average Market Share')
    plt.title('Time Series of Firm Size and Market Share')
    plt.xlabel('Year')
    plt.ylabel('Value')
    plt.legend()
    plt.show()
```
