import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the data
try:
    data = pd.read_csv("abm_pricing_data.csv") # Replace with your data file
except FileNotFoundError:
    print("Error: abm_pricing_data.csv not found.  Make sure the data file is in the correct location.")
    exit()

# Ensure 'time' column is datetime
data['time'] = pd.to_datetime(data['time'])

# Basic time series plot of average price
average_price = data.groupby('time')['price'].mean()
plt.figure(figsize=(12, 6))
plt.plot(average_price.index, average_price.values)
plt.xlabel('Time')
plt.ylabel('Average Price')
plt.title('Evolution of Average Price Over Time')
plt.show()


# Descriptive statistics
print(data['price'].describe())


# Example: Visualizing price distribution at different time points
sns.boxplot(x=data['time'].dt.year, y='price', data=data) # Aggregate to years or other suitable time scale
plt.xlabel('Year')
plt.ylabel('Price')
plt.title('Price Distribution Over Time')
plt.show()
