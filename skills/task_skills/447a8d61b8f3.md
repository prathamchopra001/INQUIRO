```python
import mesa
import numpy as np

class Firm(mesa.Agent):
    def __init__(self, unique_id, model, initial_price=10):
        super().__init__(unique_id, model)
        self.price = initial_price
        self.profit_history = []

    def step(self):
        # Simple price adjustment based on past profitability
        past_profits = np.mean(self.profit_history[-5:]) if len(self.profit_history) > 5 else 0
        if past_profits > 0:
            self.price *= 1.01  # Increase price if profitable
        else:
            self.price *= 0.99  # Decrease price if not profitable

        # Simulate selling and earning profit (replace with actual demand function)
        demand = max(0, 100 - self.price)  # Example demand function
        profit = self.price * demand
        self.profit_history.append(profit)

class MarketModel(mesa.Model):
    def __init__(self, num_firms, width, height):
        super().__init__()
        self.num_firms = num_firms
        self.grid = mesa.space.MultiGrid(width, height, torus=True)
        self.schedule = mesa.time.RandomActivation(self)

        for i in range(self.num_firms):
            a = Firm(i, self)
            self.schedule.add(a)
            x = self.random.randrange(self.grid.width)
            y = self.random.randrange(self.grid.height)
            self.grid.place_agent(a, (x, y))

    def step(self):
        self.schedule.step()

# Example run
model = MarketModel(num_firms=5, width=10, height=10)
for i in range(100): # Longer simulation
    model.step()

#Gather data for analysis (example)
all_prices = [agent.price for agent in model.schedule.agents]
print(f"Final prices: {all_prices}")
```

## Common Pitfalls
*   **Infinite Loops:** Ensure pricing adjustments don't lead to prices oscillating indefinitely. Introduce damping or constraints.
*   **Lack of Data Collection:** Implement data collection within the model to track price evolution, profits, and other relevant metrics for analysis. Use Mesa's DataCollector.
*   **Unrealistic Demand Functions:** The demand function significantly impacts results. Ensure it reflects realistic market behavior.
*   **Ignoring Spatial Effects:** If firms are located in a grid, consider how location affects competition and learning.
*   **Insufficient Simulation Length:** Run the simulation long enough to observe convergence or persistent patterns. 100 steps is a starting point; increase it significantly.
