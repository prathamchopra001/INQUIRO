```python
import numpy as np
import pandas as pd
from scipy.stats import gini
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler

def analyze_pricing_heterogeneity(data, strategy_col='pricing_strategy', size_col='firm_size', proportion_col='learning_proportion'):
    """
    Analyzes the heterogeneity of pricing strategies across different firm sizes and learning proportions.

    Args:
        data (pd.DataFrame): DataFrame with columns for firm size (small, medium, large),
                             pricing strategy (numerical representation), and the proportion of learning firms.
        strategy_col (str): Name of the column containing pricing strategies.
        size_col (str): Name of the column containing firm sizes.
        proportion_col (str): Name of the column containing the learning proportion.

    Returns:
        dict: Dictionary containing heterogeneity measures and cluster analysis results.
    """

    results = {}

    # Descriptive Statistics
    results['descriptive_stats'] = data.groupby([size_col, proportion_col])[strategy_col].describe().to_dict()

    # Heterogeneity Measures (Variance, Gini)
    results['variance'] = data.groupby([size_col, proportion_col])[strategy_col].var().to_dict()
    results['gini_coefficient'] = data.groupby([size_col, proportion_col])[strategy_col].apply(gini).to_dict()

    # Clustering (Optional)
    try:
        X = data[[strategy_col]]
        scaler = StandardScaler()
        X_scaled = scaler.fit_transform(X)
        kmeans = KMeans(n_clusters=3, random_state=42, n_init='auto') # Specify n_init explicitly
        data['cluster'] = kmeans.fit_predict(X_scaled)
        results['cluster_means'] = data.groupby('cluster')[strategy_col].mean().to_dict()
    except Exception as e:
        print(f"Clustering failed: {e}")
        results['cluster_means'] = None

    return results

# Example usage:
example_data = pd.DataFrame({
    'firm_size': ['small', 'small', 'medium', 'medium', 'large', 'large'],
    'learning_proportion': [0.2, 0.8, 0.2, 0.8, 0.2, 0.8],
    'pricing_strategy': [10, 12, 15, 18, 20, 22]
})

analysis_results = analyze_pricing_heterogeneity(example_data.copy()) # Use a copy to avoid modifying original data.
print(analysis_results)
```

## 5. Common Pitfalls

*   **Data Type Errors:** Ensure that the `pricing_strategy` column is numerical.
*   **Missing Data:** Handle missing values appropriately (imputation or removal).
*   **Incorrect Grouping:** Verify that the grouping columns (`firm_size`, `learning_proportion`) are categorical.
*   **Zero Variance:** Gini coefficient is undefined for constant values. Handle cases where variance is zero.
*   **KMeans `n_init`:** KMeans now requires explicit setting of the `n_init` parameter.  Set it to 'auto' or an integer.
*   **Modifying Original Dataframe:** When adding cluster assignments, work with a copy of the input dataframe to prevent unexpected side effects.
