def simulate_matching(num_firms, firm_sizes, num_consumers, base_match_prob):
    """
    Simulates the matching process between firms and consumers, where larger firms
    have a higher probability of matching.

    Args:
        num_firms (int): Number of firms.
        firm_sizes (list): List of firm sizes.
        num_consumers (int): Number of consumers.
        base_match_prob (float): Base matching probability.

    Returns:
        pandas.DataFrame: DataFrame containing the matching results.
    """
    firm_ids = range(num_firms)
    match_results = []

    for firm_id in firm_ids:
        # Adjust matching probability based on firm size
        adjusted_prob = base_match_prob + (firm_sizes[firm_id] / sum(firm_sizes)) * (1 - base_match_prob)
        adjusted_prob = np.clip(adjusted_prob, 0, 1) # Ensure probability is within [0, 1]

        # Simulate matching for each consumer
        matches = bernoulli.rvs(adjusted_prob, size=num_consumers)
        match_results.extend([(firm_id, consumer_id, match) for consumer_id, match in enumerate(matches)])

    return pd.DataFrame(match_results, columns=['firm_id', 'consumer_id', 'match'])

# Example usage:
num_firms = 3
firm_sizes = [10, 20, 30]
num_consumers = 100
base_match_prob = 0.1

matching_data = simulate_matching(num_firms, firm_sizes, num_consumers, base_match_prob)
print(matching_data.head())

# Analyze matching rates per firm
matching_rates = matching_data.groupby('firm_id')['match'].mean()
print(matching_rates)
