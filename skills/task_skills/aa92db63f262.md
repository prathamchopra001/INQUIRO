import numpy as np
import matplotlib.pyplot as plt
import random

class QLearningFirm:
    def __init__(self, actions, learning_rate=0.1, discount_factor=0.9, exploration_rate=0.1):
        self.q_table = {action: 0 for action in actions}
        self.actions = actions
        self.learning_rate = learning_rate
        self.discount_factor = discount_factor
        self.exploration_rate = exploration_rate

    def choose_action(self):
        if random.random() < self.exploration_rate:
            return random.choice(self.actions)
        else:
            return max(self.q_table, key=self.q_table.get)

    def update_q_table(self, action, reward, next_action):
        best_next_q = self.q_table[next_action]
        td_error = reward + self.discount_factor * best_next_q - self.q_table[action]
        self.q_table[action] += self.learning_rate * td_error

def simulate_market(num_firms, q_learning_proportion, prices, demand_function, num_episodes=1000):
    ql_firms = [QLearningFirm(prices) for _ in range(int(num_firms * q_learning_proportion))]
    random_firms = [random.choice(prices) for _ in range(num_firms - len(ql_firms))]
    all_firms = ql_firms + random_firms

    price_history = []

    for _ in range(num_episodes):
      firm_prices = []
      for firm in all_firms:
          if isinstance(firm, QLearningFirm):
              action = firm.choose_action()
              firm_prices.append(action)
          else:
              firm_prices.append(firm)

      # Calculate reward (profit) based on demand and price
      for i, firm in enumerate(all_firms):
          if isinstance(firm, QLearningFirm):
              price = firm_prices[i]
              quantity = demand_function(price, firm_prices) #Pass all prices
              reward = price * quantity
              next_action = firm.choose_action()
              firm.update_q_table(price, reward, next_action)

      price_history.append(firm_prices)

    return price_history


def demand_function(price, all_prices, max_demand=100):
    #Simple example, adjust based on your model
    average_price = np.mean(all_prices)
    if price <= average_price:
        return max_demand * (1 - (price / (2 * average_price))) # Example demand curve
    else:
        return 0


# Example Usage
num_firms = 10
prices = [1, 2, 3, 4, 5]
q_learning_proportion = 0.5
num_episodes = 1000

price_history = simulate_market(num_firms, q_learning_proportion, prices, demand_function, num_episodes)

# Calculate price dispersion (e.g., standard deviation)
last_prices = price_history[-1]
price_dispersion = np.std(last_prices)
print(f"Price Dispersion: {price_dispersion}")

## Common Pitfalls
*   **Incorrect Q-Table Initialization**: Ensure Q-values are initialized appropriately (e.g., to zero or small random values).
*   **Improper Reward Function**: The reward function must accurately reflect the firm's profit or utility. Consider market demand and competitor prices.
*   **Lack of Exploration**:  Firms may converge to suboptimal strategies if the exploration rate is too low.
*   **Incorrect indexing**: When passing prices to reward functions, ensure the indexes match the correct firms.
*   **Passing only one price to demand**:  Ensure the demand function receives all prices in the market.
