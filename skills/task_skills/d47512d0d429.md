## Example Code
```python
import pandas as pd
import numpy as np
from scipy.stats import describe

# Load data (replace 'simulation_output.csv' with your actual filename)
try:
    df = pd.read_csv('simulation_output.csv')
except FileNotFoundError:
    print("Error: simulation_output.csv not found.  Ensure the file exists in the correct directory.")
    exit()

# Ensure 'firm_size' and 'pricing_strategy' columns exist
if 'firm_size' not in df.columns or 'pricing_strategy' not in df.columns:
    print("Error: 'firm_size' or 'pricing_strategy' column not found in the CSV.")
    exit()

# Group by firm size and pricing strategy, then count occurrences
grouped = df.groupby(['firm_size', 'pricing_strategy']).size().reset_index(name='counts')

# Calculate descriptive statistics for pricing strategies within each firm size
def describe_pricing(group):
    desc = describe(group['counts'])
    return pd.Series({
        'nobs': desc.nobs,
        'mean': desc.mean,
        'variance': desc.variance,
        'skewness': desc.skewness,
        'kurtosis': desc.kurtosis
    })

descriptive_stats = grouped.groupby('firm_size').apply(describe_pricing)

print(descriptive_stats)

# Example: Pivot table for a clearer view (optional)
pivot_table = grouped.pivot_table(index='firm_size', columns='pricing_strategy', values='counts', fill_value=0)
print("\nPivot Table (Firm Size vs. Pricing Strategy Counts):\n", pivot_table)
```

*   **Incorrect Grouping:** Verify that the `groupby()` columns are the intended grouping variables.  Incorrect grouping will lead to misleading results.
