import mesa
import numpy as np

class Agent(mesa.Agent):
    def __init__(self, unique_id, model):
        super().__init__(unique_id, model)

    def step(self):
        # Simple random pricing strategy
        self.price = np.random.uniform(0, 10) # Example pricing
        # Basic interaction with market
        self.demand = self.model.get_demand(self.price) # Assume market demand calculation
        self.revenue = self.price * self.demand
        #print(f"Agent {self.unique_id}: Price={self.price:.2f}, Demand={self.demand:.2f}, Revenue={self.revenue:.2f}")


class MarketModel(mesa.Model):
    def __init__(self, num_agents, demand_variability):
        self.num_agents = num_agents
        self.demand_variability = demand_variability
        self.schedule = mesa.space.SingleGrid(width=10, height=10, torus=True) #Example grid

        self.schedule = mesa.time.RandomActivation(self)

        for i in range(self.num_agents):
            a = Agent(i, self)
            self.schedule.add(a)
            self.schedule.place_agent(a, (i % 10, i // 10))


    def step(self):
        self.schedule.step()

    def get_demand(self, price):
        # Simplified demand function (example)
        base_demand = 100 - price
        noise = np.random.normal(0, self.demand_variability)
        return max(0, base_demand + noise)  # Ensure demand is non-negative


# Run the simulation
model = MarketModel(num_agents=5, demand_variability=5)
for i in range(10):
    model.step()
