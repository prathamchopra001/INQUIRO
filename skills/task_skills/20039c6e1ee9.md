```python
import mesa
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

class Firm(mesa.Agent):
    def __init__(self, unique_id, model):
        super().__init__(unique_id, model)
        self.price = 10  # Initial price
        self.learning_rate = 0.1

    def step(self):
        # Simple price adjustment based on market signal (example)
        market_signal = self.model.get_market_signal()
        self.price += self.learning_rate * (market_signal - self.price)

    def get_price(self):
        return self.price


class MonetaryPolicyModel(mesa.Model):
    def __init__(self, num_firms, policy_regime="interest_rate"):
        super().__init__()
        self.num_firms = num_firms
        self.policy_regime = policy_regime
        self.schedule = mesa.space.MultiGrid(10, 10, False) # Example space
        self.datacollector = mesa.DataCollector(
            model_reporters={"AvgPrice": lambda m: np.mean([a.get_price() for a in m.schedule.agents])},
            agent_reporters={"Price": "price"} # Access price directly
        )


        # Create firms
        for i in range(self.num_firms):
            firm = Firm(i, self)
            self.schedule.place_agent(firm, (i % 10, i // 10)) # Example placement

        self.running = True
        self.datacollector.collect(self)


    def get_market_signal(self):
        # Simulate a market signal based on the policy regime
        if self.policy_regime == "interest_rate":
            return 12  # Example interest rate signal
        elif self.policy_regime == "inflation_targeting":
            return 11 # Example inflation signal
        else:
            return 10

    def step(self):
        self.schedule.step()
        self.datacollector.collect(self)

# Run the simulation
model = MonetaryPolicyModel(num_firms=10, policy_regime="inflation_targeting")
for i in range(10):
    model.step()

# Analyze results
results = model.datacollector.get_model_vars_dataframe()
print(results.head())

# Example plot
results.plot()
plt.show()
```
