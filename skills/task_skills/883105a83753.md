import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

class Firm:
    def __init__(self, size):
        self.size = size
        self.price = 1.0  # Initial price

    def adjust_price(self, market_price):
        # Simple price adjustment based on market price
        self.price = 0.9 * self.price + 0.1 * market_price


def simulate(num_firms, time_steps, size_dependent_matching=True):
    firms = [Firm(np.random.randint(1, 10)) for _ in range(num_firms)] #Random size between 1-9
    prices = []

    for _ in range(time_steps):
        # Matching process (simplified)
        if size_dependent_matching:
            # Size-dependent matching probability (example: larger firms match more often)
            match_probs = [firm.size / sum(f.size for f in firms) for firm in firms]
        else:
            # Size-independent matching (uniform probability)
            match_probs = [1/num_firms] * num_firms


        matched_firm_index = np.random.choice(num_firms, p=match_probs)
        market_price = firms[matched_firm_index].price #Simplification: The matched firm dictates the market price

        for firm in firms:
            firm.adjust_price(market_price)
        prices.append([firm.price for firm in firms])

    return np.array(prices)

# Example usage
num_firms = 50
time_steps = 100

prices_size_dependent = simulate(num_firms, time_steps, size_dependent_matching=True)
prices_size_independent = simulate(num_firms, time_steps, size_dependent_matching=False)

# Basic plotting (expand this for detailed analysis)
plt.plot(prices_size_dependent.mean(axis=1), label='Size Dependent')
plt.plot(prices_size_independent.mean(axis=1), label='Size Independent')
plt.xlabel('Time Step')
plt.ylabel('Average Price')
plt.legend()
plt.show()
