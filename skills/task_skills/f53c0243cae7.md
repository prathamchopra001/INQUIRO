class PricingEnv(gym.Env):
    def __init__(self, n_agents=2, max_price=10, cost=1):
        super(PricingEnv, self).__init__()
        self.n_agents = n_agents
        self.max_price = max_price
        self.cost = cost
        self.observation_space = spaces.Tuple([spaces.Discrete(max_price + 1) for _ in range(n_agents)]) #Agent's own price
        self.action_space = spaces.Discrete(max_price + 1) #Each agent sets its own price

        self.prices = [0] * n_agents # Initialize prices

    def reset(self, seed=None, options=None):
        super().reset(seed=seed)
        self.prices = [np.random.randint(1, self.max_price + 1) for _ in range(self.n_agents)]  #Initial random prices
        observations = tuple(self.prices) #Each agent observes its own initial price
        info = {}
        return observations, info

    def step(self, actions):
        #Actions are the prices chosen by each agent
        self.prices = actions
        rewards = self._calculate_rewards()
        observations = tuple(self.prices) #Each agent observes its own current price
        terminated = False #No terminal state
        truncated = False
        info = {}
        return observations, rewards, terminated, truncated, info

    def _calculate_rewards(self):
        rewards = []
        for i in range(self.n_agents):
            profit = 0
            my_price = self.prices[i]
            demand = 1 #Assume demand is 1 if my price is the lowest
            for j in range(self.n_agents):
                if i != j and self.prices[j] < my_price:
                    demand = 0 #Demand is zero if someone undercut me

            profit = (my_price - self.cost) * demand
            rewards.append(profit)
        return rewards

#Example usage
env = PricingEnv()
obs, info = env.reset()
actions = [env.action_space.sample() for _ in range(env.n_agents)]
obs, rewards, terminated, truncated, info = env.step(actions)
print("Observations:", obs)
print("Rewards:", rewards)
```

## Common Pitfalls
*   **Incorrect Reward Calculation:** Ensure the reward function accurately reflects the profit based on price competition and cost. Specifically, remember to zero out the demand if a competitor has a lower price.
*   **State Representation:** Agents should only observe their *own* price (asymmetric information). The observation space should reflect this.
*   **Exploration vs. Exploitation:** Implement an appropriate exploration strategy (e.g., epsilon-greedy) in the RL agent to ensure sufficient exploration of the price space.
*   **Environment Reset:** Properly reset the environment at the beginning of each episode, including randomizing initial prices.
*   **Discrete Action Space:** Using `spaces.Discrete` requires integer actions. Ensure actions are cast to integers if necessary.
