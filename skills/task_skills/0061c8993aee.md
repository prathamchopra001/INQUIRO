class SimpleModel(mesa.Model):
    def __init__(self, N, initial_productivity):
        self.num_agents = N
        self.schedule = mesa.time.RandomActivation(self)
        for i in range(self.num_agents):
            a = SimpleAgent(i, self, initial_productivity)
            self.schedule.add(a)
        self.running = True

    def step(self):
        self.schedule.step()

    def get_gdp(self):
        # Simulate GDP calculation
        return sum([agent.productivity for agent in self.schedule.agents])

def objective_function(params, model_params, target_gdp):
    """Calculates the difference between simulated and target GDP."""
    N, initial_productivity = model_params
    model = SimpleModel(N, params[0]) # Calibrating initial productivity
    for _ in range(10):  # Run model for 10 steps
        model.step()
    simulated_gdp = model.get_gdp()
    return (simulated_gdp - target_gdp)**2

# Calibration Setup
N = 10

## Common Pitfalls
*   **Non-Stationary Time Series:** Ensure time series data (GDP, Inflation) is stationary before using it for calibration. Use `adfuller` test from `statsmodels`.  Consider differencing if needed.
*   **Incorrect Objective Function:** The objective function *must* quantify the difference between model output and real-world data. Carefully choose a metric (e.g., Mean Squared Error, Mean Absolute Percentage Error).  The example uses a squared error, but this should be adapted to the specific data and goals.
*   **Ignoring Parameter Bounds:** Define reasonable bounds for parameters during optimization (using `bounds` in `minimize`). This prevents the optimizer from exploring unrealistic parameter values.
*   **Insufficient Model Runtime:** Ensure the ABM runs for a sufficient number of steps to capture the relevant dynamics.
*   **Lack of Validation:** After calibration, validate the model against other macroeconomic indicators not used in the calibration process.
*   **Overfitting:** Avoid calibrating too many parameters with limited data, which can lead to overfitting.
