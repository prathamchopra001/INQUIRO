import numpy as np
import pandas as pd

class Firm(mesa.Agent):
    def __init__(self, unique_id, model, size):
        super().__init__(unique_id, model)

class ABM(mesa.Model):
    def __init__(self, N, width, height):
        self.num_agents = N
        self.grid = mesa.space.MultiGrid(width, height, True)
        self.schedule = mesa.time.RandomActivation(self)
        self.total_size = 0 # Keep track of total firm size
        # Create agents
        for i in range(self.num_agents):
            size = np.random.randint(1, 10)  # Firm size
            a = Firm(i, self, size)
            self.total_size += size
            self.schedule.add(a)
            # Add the agent to a random grid cell
            x = self.random.randrange(self.grid.width)
            y = self.random.randrange(self.grid.height)
            self.grid.place_agent(a, (x, y))
        self.datacollector = mesa.DataCollector(
            model_reporters={"Total Profit": lambda m: sum([a.profit for a in m.schedule.agents])},
            agent_reporters={"Size": "size", "Profit": "profit", "Price": "price"},
        )


    def step(self):
        '''Advance the model by one step.'''
        self.datacollector.collect(self)
        self.schedule.step()

# Run the model
model = ABM(N=10, width=10, height=10)
for i in range(10):
    model.step()

# Collect results
results = model.datacollector.get_agent_vars_dataframe()

## Common Pitfalls
*   **Incorrect Probability Calculation:** Ensure matching probabilities are between 0 and 1.
*   **Ignoring Agent Type:**  When matching, verify that the agent you are trying to match with is the correct type (e.g., another Firm).
*   **Data Collection Errors:**  Double-check that data is being collected correctly using `DataCollector`.  Specifically, ensure your model and agent reporters are using the correct names for properties.
*   **Division by Zero:** If calculating market share, handle the case where total market size is zero.
*   **Unintended self-interaction:** Ensure an agent cannot match with itself.
