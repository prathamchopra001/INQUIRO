class PricingEnv(gym.Env):
    def __init__(self, reward_function='profit'):
        super(PricingEnv, self).__init__()
        self.action_space = gym.spaces.Discrete(3)  # Example: Low, Medium, High price
        self.observation_space = gym.spaces.Discrete(1) # Simplification: Single state
        self.reward_function = reward_function

    def step(self, action):
        # Simulate demand based on action (price)
        if action == 0: demand = 10  # Low price
        elif action == 1: demand = 7   # Medium price
        else: demand = 3                # High price

        cost = 2  # Cost per unit
        price = [5, 7, 10][action] # Corresponding prices

        profit = (price - cost) * demand
        market_share = demand / 20 # Assuming max possible demand is 20

        if self.reward_function == 'profit':
            reward = profit
        elif self.reward_function == 'market_share':
            reward = market_share
        else:
            reward = profit # Default to profit

        return 0, reward, True, {}  # Simplified: Always terminal state

    def reset(self):
        return 0

# Dummy RL training loop (replace with actual RL algorithm)
